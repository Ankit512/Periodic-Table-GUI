
DROP database if exists Elements;
create database Elements;
use Elements;
create table Data(AtomicNumber integer(3) primary key, 
Symbol varchar(3) UNIQUE, 
Name varchar(16) UNIQUE, 
GroupNo varchar(12) NOT NULL, 
Period integer(1) NOT NULL, 
AtomicMass double(6,3),
Configuration varchar(25),
Source varchar(45),
Characteristics varchar(35),
Uses varchar(30))
engine = innodb;

create table Properties(AtomicNumber integer(3) primary key references AtomicNumber(Data),
Symbol varchar(3) REFERENCES Symbol(Data),
Name varchar(16) REFERENCES Name(Data),
MeltingPoint double(6,2),
BoilingPoint double(6,2),
Density double(4,2),
SpecificHeat double(6,3),
HeatOfFusion double(6,3),
HeatOfVapourisation double(6,3))
engine = innodb;
